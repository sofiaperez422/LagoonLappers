# read in your packages here
library(tidyverse) # general use
library(here) # file organization
library(janitor) # cleaning data frames
library(readxl) # reading excel files
library(scales) # modifying axis labels
library(ggeffects) # getting model predictions
library(MuMIn) # model selection
library(showtext) # to show custom text font on plots
library(flextable) # create tables for model summaries

font_add_google(name = "Public Sans", # name of font on Google font website
              family = "Psans") # name to call in font as
showtext_auto()

# read in your data here
labrynth <- read_csv(here("data",
                             "LagoonLappers_DataSheetES153 - R-Studio Data.csv"))
# clean data
labrynth_clean <- labrynth |> # create new object
  clean_names() # clean column names to lower case and underscores

Figure 1. Sunlight Exposure and Flower Count

ggplot(data = labrynth_clean,
       aes(x = shadow_from_tree_root_ft,
       y = flower_bud_count)) +
  geom_ribbon(data = model_pred1, # plot prediction 95% CI using mod_preds object
              aes(x = x, # use same x-axix
                  y = predicted, # set y-axis as predicted values
                  ymin = conf.low, # set bottom of region
                  ymax = conf.high), # set top
              alpha = 0.2, # adjust tranparency
              fill = "firebrick2") + # change color
  geom_line(data = model_pred1, # plot model prediction as line using mod_preds object
            aes(x = x, # use same x-axis
                y = predicted), # set y-axis as predicted values
            linewidth = 1, # change line width to thicker 
            color = "firebrick4") + # change color
  geom_point(size = 3.5,
             alpha = 0.5,
             aes(color = common_name_plant)) +
  
  
  labs(title = "Flower bud patterns by tree shadow",
       x = "Shadow length from tree root (ft)",
       y = "Flower bud count",
       color = "Species") +
  theme_light(base_family = "Psans") + # change theme and font
  theme(legend.position = "right",
        plot.title = element_text(hjust = 0.5))
